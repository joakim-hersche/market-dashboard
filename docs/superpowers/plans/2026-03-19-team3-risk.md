def _render_sector_breakdown(portfolio_df, fund_rows: list[dict], portfolio_color_map: dict) -> None:
    """Render sector exposure as horizontal bar chart."""
    with ui.column().classes("chart-card w-full"):
        ui.label("Sector Exposure").classes("chart-title")

        # Build sector weights
        sector_weights = {}
        for row in fund_rows:
            ticker = row.get("Ticker", "")
            sector = row.get("Sector", "Unknown")
            # Get weight from portfolio_df
            weight_rows = portfolio_df[portfolio_df["Ticker"] == ticker] if not portfolio_df.empty else pd.DataFrame()
            weight = weight_rows["Share (%)"].sum() if not weight_rows.empty else 0
            sector_weights[sector] = sector_weights.get(sector, 0) + weight

        if not sector_weights:
            ui.label("No sector data available.").style(f"color: {TEXT_FAINT};")
            return

        # Sort by weight descending
        sorted_sectors = sorted(sector_weights.items(), key=lambda x: x[1], reverse=True)

        # Neutral palette for sectors
        sector_colors = [
            "#3B82F6", "#6366F1", "#8B5CF6", "#14B8A6", "#06B6D4",
            "#0EA5E9", "#7C3AED", "#2563EB", "#0891B2", "#0D9488",
            "#4F46E5", "#1D4ED8",
        ]

        bars_html = []
        for i, (sector, weight) in enumerate(sorted_sectors):
            color = sector_colors[i % len(sector_colors)]
            bar_width = max(weight, 0.5)  # minimum visible width
            bars_html.append(
                f'<div style="display:flex;align-items:center;gap:12px;margin-bottom:6px;">'
                f'<div style="width:120px;text-align:right;color:{TEXT_SECONDARY};font-size:12px;flex-shrink:0;">{sector}</div>'
                f'<div style="flex:1;position:relative;height:22px;">'
                f'<div class="alloc-bar" style="width:{bar_width}%;height:100%;background:{color};border-radius:4px;position:relative;">'
                f'<span style="position:absolute;right:8px;top:50%;transform:translateY(-50%);color:white;font-size:11px;font-weight:600;">{weight:.1f}%</span>'
                f'</div></div></div>'
            )

        ui.html(f'<div style="padding:8px 0;">{"".join(bars_html)}</div>')
```

def _render_rebalancing_calculator(portfolio_df, currency: str) -> None:
    """Render interactive rebalancing calculator with buy-only suggestions."""
    with ui.column().classes("chart-card w-full"):
        ui.label("Rebalancing Helper").classes("chart-title")
        ui.label("Buy-only — does not suggest sells").style(f"color: {TEXT_FAINT}; font-size: 12px; margin-bottom: 12px;")

        if portfolio_df.empty:
            ui.label("Add positions to use the rebalancing calculator.").style(f"color: {TEXT_FAINT};")
            return

        # Group by ticker to get current weights
        ticker_weights = portfolio_df.groupby("Ticker").agg({
            "Share (%)": "sum",
            "Current Price": "first",
            "Total Value": "sum",
        }).reset_index()

        # Target weight inputs
        target_inputs = {}
        deposit_input = ui.number("Next deposit", value=0, min=0, step=100, suffix=currency).classes("w-48")

        # Build input table
        with ui.column().classes("w-full gap-2 mt-4"):
            # Header row
            with ui.row().classes("w-full items-center gap-4"):
                ui.label("Ticker").style(f"width:80px;color:{TEXT_FAINT};font-size:11px;text-transform:uppercase;")
                ui.label("Current %").style(f"width:80px;color:{TEXT_FAINT};font-size:11px;text-transform:uppercase;text-align:right;")
                ui.label("Target %").style(f"width:100px;color:{TEXT_FAINT};font-size:11px;text-transform:uppercase;")
                ui.label("Drift").style(f"width:80px;color:{TEXT_FAINT};font-size:11px;text-transform:uppercase;text-align:right;")
                ui.label("Action").style(f"flex:1;color:{TEXT_FAINT};font-size:11px;text-transform:uppercase;")

            # Result container
            result_container = ui.column().classes("w-full")

            for _, row in ticker_weights.iterrows():
                ticker = row["Ticker"]
                current_pct = row["Share (%)"]
                target_inputs[ticker] = {
                    "current": current_pct,
                    "price": row["Current Price"],
                    "input": ui.number(value=round(current_pct, 1), min=0, max=100, step=0.5).classes("w-24"),
                }

        async def _update_rebalance():
            result_container.clear()
            deposit = deposit_input.value or 0
            total_target = sum(inp["input"].value or 0 for inp in target_inputs.values())

            with result_container:
                if abs(total_target - 100) > 0.5:
                    ui.label(f"Target weights sum to {total_target:.1f}% (should be ~100%)").style(f"color: {AMBER}; font-size: 12px;")

                # Calculate drift and allocate deposit
                drifts = []
                for ticker, inp in target_inputs.items():
                    target = inp["input"].value or 0
                    current = inp["current"]
                    drift = current - target
                    drifts.append({"ticker": ticker, "current": current, "target": target, "drift": drift, "price": inp["price"]})

                # Sort: most underweight first
                drifts.sort(key=lambda d: d["drift"])

                remaining = deposit
                actions = {}
                if deposit > 0:
                    for d in drifts:
                        if d["drift"] >= 0 or d["price"] <= 0:
                            continue
                        # How much money needed to reach target
                        shares_to_buy = int(remaining / d["price"]) if d["price"] > 0 else 0
                        if shares_to_buy > 0:
                            cost = shares_to_buy * d["price"]
                            remaining -= cost
                            actions[d["ticker"]] = (shares_to_buy, cost)

                for d in drifts:
                    drift_color = GREEN if abs(d["drift"]) < 2 else (AMBER if abs(d["drift"]) < 5 else RED)
                    action_text = ""
                    if d["ticker"] in actions:
                        shares, cost = actions[d["ticker"]]
                        action_text = f"Buy {shares} shares (~{currency} {cost:,.0f})"
                    elif d["drift"] > 2:
                        action_text = f"Overweight by {d['drift']:.1f}%"
                    elif deposit == 0:
                        action_text = "Enter a deposit amount to see buy suggestions"

                    with ui.row().classes("w-full items-center gap-4").style(f"border-bottom:1px solid rgba(255,255,255,0.06);padding:6px 0;"):
                        ui.label(d["ticker"]).style(f"width:80px;color:{TEXT_SECONDARY};font-size:13px;")
                        ui.label(f"{d['current']:.1f}%").style(f"width:80px;color:{TEXT_SECONDARY};font-size:13px;text-align:right;")
                        ui.label(f"{d['target']:.1f}%").style(f"width:100px;color:{TEXT_SECONDARY};font-size:13px;")
                        ui.label(f"{d['drift']:+.1f}%").style(f"width:80px;color:{drift_color};font-size:13px;text-align:right;")
                        ui.label(action_text).style(f"flex:1;color:{TEXT_FAINT};font-size:12px;")

        # Wire up reactive updates
        deposit_input.on_value_change(lambda _: _update_rebalance())
        for inp in target_inputs.values():
            inp["input"].on_value_change(lambda _: _update_rebalance())

        # Initial render
        await _update_rebalance()
```
