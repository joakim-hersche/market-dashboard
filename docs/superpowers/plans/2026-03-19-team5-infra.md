```python
"""Tests for src/portfolio.py core functions."""
import pandas as pd
from unittest.mock import patch, MagicMock


@patch("src.portfolio.yf.Ticker")
@patch("src.portfolio.get_fx_rate", return_value=(1.0, True))
@patch("src.portfolio.get_historical_fx_rate", return_value=(1.0, True))
@patch("src.portfolio.fetch_buy_price", return_value=(100.0, "2025-01-15"))
def test_build_portfolio_df_basic(mock_buy, mock_hist_fx, mock_fx, mock_ticker):
    from src.portfolio import build_portfolio_df

    # Mock current price
    mock_ticker.return_value.info = {"currentPrice": 110.0, "shortName": "Apple Inc"}
    mock_ticker.return_value.history.return_value = pd.DataFrame(
        {"Dividends": [0.0]}, index=pd.DatetimeIndex(["2025-01-15"])
    )

    portfolio = {
        "AAPL": [{"shares": 10, "buy_price": 100.0, "buy_fx_rate": 1.0, "purchase_date": "2025-01-15", "manual_price": False}]
    }

    # Clear cache to avoid stale results
    build_portfolio_df.cache_clear() if hasattr(build_portfolio_df, 'cache_clear') else None

    result = build_portfolio_df(portfolio, "USD")
    assert isinstance(result, pd.DataFrame)
    assert len(result) > 0
    assert "Ticker" in result.columns
    assert "Total Value" in result.columns


def test_build_portfolio_df_empty():
    from src.portfolio import build_portfolio_df
    result = build_portfolio_df({}, "USD")
    assert isinstance(result, pd.DataFrame)
    assert len(result) == 0
```

```python
"""Tests for src/ui/shared.py — encryption round-trip."""
import json


def test_encryption_round_trip():
    from src.ui.shared import encrypt_portfolio, decrypt_portfolio

    original = {
        "portfolio": {"AAPL": [{"shares": 10, "buy_price": 150.0}]},
        "currency": "USD",
    }
    encrypted = encrypt_portfolio(original)
    assert encrypted != json.dumps(original)  # Should be different (encrypted)

    decrypted = decrypt_portfolio(encrypted)
    assert decrypted == original


def test_decrypt_unencrypted_data():
    """Legacy unencrypted data should be handled gracefully."""
    from src.ui.shared import decrypt_portfolio

    raw = {"portfolio": {"AAPL": [{"shares": 5}]}, "currency": "EUR"}
    result = decrypt_portfolio(raw)
    assert result == raw  # Should pass through unchanged
```
